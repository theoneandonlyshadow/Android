              package com.example.textileshopbilling;
              import android.content.ContentValues;
              import android.content.Context;
              import android.database.Cursor;
              import android.database.sqlite.SQLiteDatabase;
              import android.database.sqlite.SQLiteOpenHelper;
              public class DatabaseHelper extends SQLiteOpenHelper {
              // Database name and table details
              private static final String DATABASE_NAME = "TextileShop.db";
              private static final String TABLE_NAME = "billing";
              private static final String COL_BILL_ID = "bill_id";
              private static final String COL_CUSTOMER_NAME = "customer_name";
              private static final String COL_MOBILE = "mobile_number";
              private static final String COL_AMOUNT = "total_amount";
              public DatabaseHelper(Context context) {
              super(context, DATABASE_NAME, null, 1);
              }
              @Override
              public void onCreate(SQLiteDatabase db) {
              String createTable = "CREATE TABLE " + TABLE_NAME + " (" +
              COL_BILL_ID + " INTEGER PRIMARY KEY AUTOINCREMENT, " +
              COL_CUSTOMER_NAME + " TEXT, " +
              COL_MOBILE + " TEXT, " +
              COL_AMOUNT + " REAL)";
              db.execSQL(createTable);
              }
              @Override
              public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
              db.execSQL("DROP TABLE IF EXISTS " + TABLE_NAME);
              onCreate(db);
              }
              // Insert billing details
              public boolean insertBillingDetails(String name, String mobile, double amount) {
              SQLiteDatabase db = this.getWritableDatabase();
              ContentValues values = new ContentValues();
              values.put(COL_CUSTOMER_NAME, name);
              values.put(COL_MOBILE, mobile);
              values.put(COL_AMOUNT, amount);
              long result = db.insert(TABLE_NAME, null, values);
              return result != -1; // Return true if insert is successful
              }
              // Fetch all billing details
              public Cursor getAllBillingDetails() {
              SQLiteDatabase db = this.getReadableDatabase();
              return db.rawQuery("SELECT * FROM " + TABLE_NAME, null);
              }
              // Get total amount billed
              public Cursor getTotalAmountBilled() {
              SQLiteDatabase db = this.getReadableDatabase();
              return db.rawQuery("SELECT SUM(" + COL_AMOUNT + ") AS TotalAmount FROM " + TABLE_NAME, null);
              }
              // Get details by bill number
              public Cursor getBillingByBillNumber(int billId) {
              SQLiteDatabase db = this.getReadableDatabase();
              return db.rawQuery("SELECT * FROM " + TABLE_NAME + " WHERE " + COL_BILL_ID + " = ?", new
              String[]{String.valueOf(billId)});
              }
              // Get total number of customers
              public Cursor getTotalCustomerCount() {
              SQLiteDatabase db = this.getReadableDatabase();
              return db.rawQuery("SELECT COUNT(DISTINCT " + COL_MOBILE + ") AS TotalCustomers FROM " + TABLE_NAME, null);
              }
              }

              package com.example.end2;
  
              import android.content.ContentValues;
              import android.content.Context;
              import android.database.Cursor;
              import android.database.sqlite.SQLiteDatabase;
              import android.database.sqlite.SQLiteOpenHelper;
  
              public class DBHelper extends SQLiteOpenHelper {
                  private static final String DATABASE_NAME = "BikeShowroom.db";
                  private static final int DATABASE_VERSION = 1;
  
                  public DBHelper(Context context) {
                      super(context, DATABASE_NAME, null, DATABASE_VERSION);
                  }
  
                  @Override
                  public void onCreate(SQLiteDatabase db) {
                      String createTable = "CREATE TABLE Bookings (" +
                                           "customer_name TEXT, " +
                                           "bank_name TEXT, " +
                                           "amount_paid REAL, " +
                                           "transaction_id TEXT PRIMARY KEY)";
                      db.execSQL(createTable);
                  }
  
                  @Override
                  public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
                      db.execSQL("DROP TABLE IF EXISTS Bookings");
                      onCreate(db);
                  }
  
                  public boolean insertBooking(String customerName, String bankName, double amountPaid, String transactionId) {
                      SQLiteDatabase db = this.getWritableDatabase();
                      ContentValues values = new ContentValues();
                      values.put("customer_name", customerName);
                      values.put("bank_name", bankName);
                      values.put("amount_paid", amountPaid);
                      values.put("transaction_id", transactionId);
                      long result = db.insert("Bookings", null, values);
                      return result != -1;
                  }
  
                  public Cursor getAllBookings() {
                      SQLiteDatabase db = this.getReadableDatabase();
                      return db.rawQuery("SELECT * FROM Bookings", null);
                  }
  
                  public double getTotalAmount() {
                      SQLiteDatabase db = this.getReadableDatabase();
                      Cursor cursor = db.rawQuery("SELECT SUM(amount_paid) FROM Bookings", null);
                      if (cursor.moveToFirst()) {
                          return cursor.getDouble(0);
                      }
                      return 0.0;
                  }
  
                  public Cursor getBookingByTransactionId(String transactionId) {
                      SQLiteDatabase db = this.getReadableDatabase();
                      return db.rawQuery("SELECT customer_name, amount_paid FROM Bookings WHERE transaction_id = ?",
                              new String[]{transactionId});
                  }
  
                  public int getTotalBikesBooked() {
                      SQLiteDatabase db = this.getReadableDatabase();
                      Cursor cursor = db.rawQuery("SELECT COUNT(*) FROM Bookings", null);
                      if (cursor.moveToFirst()) {
                          return cursor.getInt(0);
                      }
                      return 0;
                  }
              }

                                  int distance = getDestinationDistance(getIntent().getStringExtra("DESTINATION"));
                                  double pricePerKm = getPricePerKm(selectedClass);
                                  double ticketPrice = calculateTicketPrice(distance, pricePerKm, ageValue);

                                  ticketDetails.setText(details);
                              }

                      private int getDestinationDistance(String destination) {
                          switch (destination) {
                              case "Bangalore": return 200;
                              case "Mysore": return 150;
                              case "Chennai": return 350;
                              default: return 0;
                          }
                      }
                      private double getPricePerKm(String selectedClass) {
                          switch (selectedClass) {
                              case "General": return 1.5;
                              case "Sleeper": return 3.0;
                              case "AC": return 6.0;
                              default: return 0;
                          }
                      }
                      private double calculateTicketPrice(int distance, double pricePerKm, int age) {
                          double ticketPrice = distance * pricePerKm;
                          if (age >= 60) {
                              ticketPrice /= 2; // 50% discount for senior citizens
                          }
                          return ticketPrice;
                      }
